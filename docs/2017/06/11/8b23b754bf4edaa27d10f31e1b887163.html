草，最后终于搞定了。
先通过which go找到，go的可执行文件/usr/bin/go，然后ll /usr/bin/go发现它指向另外一个目录，/usr/lib/go-1.7，于是在~/.bashrc中加入

export GOROOT=/usr/lib/go-1.7
export GOPATH=$HOME/gopath
export PATH=$PATH:$GOROOT/bin:$GOPATH:bin

然后source ~/.bashrc
