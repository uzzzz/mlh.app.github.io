+ (unsigned short)crc16:(unsigned char *)puchMsg andLength:(unsigned short)usDataLen

{

    unsigned char uchCRCHi = 0xFF ; /* 初始化高字节*/

    unsigned char uchCRCLo = 0xFF ; /* 初始化低字节*/

    unsigned uIndex ; /*把CRC表*/

    while (usDataLen--) /*通过数据缓冲器*/

    {

        uIndex = uchCRCHi ^ *(puchMsg++) ; /*计算CRC */

        uchCRCHi = uchCRCLo ^ auchCRCHi[uIndex] ;

        uchCRCLo = auchCRCLo[uIndex] ;

    }

    return (uchCRCHi << 8 | uchCRCLo) ;

}

+ (unsigned short)crc16: (NSData*) data

{

    unsigned int    crc;

    crc = 0x4846;  

    

    uint8_t byteArray[[data length]];

    [data getBytes:&byteArray length:sizeof(byteArray)];

    

    for (int i = 0; i<[data length]; i++) {

        Byte byte = byteArray[i];

        crc = (crc >> 8) ^ crc16table[(crc^ byte) & 0xFF];

    }

    return crc;

}
