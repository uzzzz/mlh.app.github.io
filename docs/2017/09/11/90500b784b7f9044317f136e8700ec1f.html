        unsigned short cal2_crc16(unsigned char *ptr, short len)
        {
        unsigned short crc = 0;
        unsigned char da = 0;

        crc = 0;
        while(len-- != 0)
        {
        da = (unsigned char)(crc / 256);// 以8位二进制数的形式暂存CRC的高8位 
        crc <<= 8;        // 左移8位,相当于CRC的低8位乘以28 
        crc ^= crc_ta[ da ^ *ptr ]; // 高8位和当前字节相加后再查表求CRC,再加上以前的CRC 
        ptr++;

        }
  /* */
        return (crc);
        }

    byte da;
    crc = 0x0;

        da = (byte) (crc / 256);
        crc <<= 8;
