一 代码

  1. import rsa
  3. key = rsa.newkeys(3000)#生成随机秘钥
  4. privateKey = key[1]#私钥
  5. publicKey = key[0]#公钥
  7. message ='sanxi Now is better than never.'
  8. print('Before encrypted:',message)
  9. message = message.encode()
  11. cryptedMessage = rsa.encrypt(message, publicKey)
  12. print('After encrypted:\n',cryptedMessage)
  14. message = rsa.decrypt(cryptedMessage, privateKey)
  15. message = message.decode()
  16. print('After decrypted:',message)
