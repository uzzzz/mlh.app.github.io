2.工具准备及安装
2.1简单的安装ssh，安装git，安装pip

sudo apt-get install ssh

sudo apt-get install git

sudo apt-get install python-pip

2.2安装go

uname -a                             //查看ubuntu系统详细，确定go版本版本--（很简单 i686，i386表示是32位，x64表示64位）

                                     //下载go语言包 https://golang.org/dl/  下载网址，（听说要翻墙，当然可以看其他教程）

tar -zvxf go1.10.3.linux-386.tar.gz  //解压

mv go /usr/local/ 		             //解压之后的go文件夹移动到 /usr/local/ 目录下面

vi /etc/profile                      //配置环境变量（文件末尾加上这三句），所有环境变量配置三句：
export GOROOT=/usr/local/go    #你的go语言包的位置
export PATH=$PATH:/usr/local/go/bin #添加go语言包的bin到path变量里
export GOPATH=/root/gopath  #go的项目存放地址，以后你的go项目需要放在哪里，这个你自己随意设置，重要的是上面两个

source /etc/profile                  //重新应用你的配置文件

go version                            //这个时候查看go 版本，应该已经成功了
