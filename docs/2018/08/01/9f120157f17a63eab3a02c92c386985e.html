整型：

int/uint

int——有符号的整型

uint——无符号的整型

int8-int256

<= < == != >= >

& | ^ ~

+ - 一元+ 一元0- * / % ** << >>

pragma solidity ^0.4.0;

 

contract Test{

    uint a;

    int b = 20;

    int c = 30;

    function testadd() public constant returns(int){

        if(b > c){

            return b + c;

        } else if(b == c){

            return b * c;

        } else{

            return b >> 2;

        }

        -1 >> 2;

    }

}
