unsigned short table_crc(unsigned char*ptr, unsigned short len, unsigned short crc)
{
    unsigned char temp;

    if(0 == len)
        return crc;
    while(len-- != 0)
    {
        temp  = (unsigned char)(crc/256); // 以8 位二进制数暂存CRC 的高8 位
        crc<<= 8;// 左移8 位
        crc^= crc_table[temp^(*ptr)]; // 高字节和当前数据XOR 再查表
        ptr++;
    }
    return(crc);
}
