金蝶K3提取供应商信息生成在线Web页面供应商信息一览表

  • 柚子社区
1、IMS创建供应商信息一览表表单，后台表名称为T7DQ48L；

2、将IMS数据库供应商信息一览表重命名；

3、提取并封装K3供应商数据视图。代码如下：

USE [portal]
GO

/****** Object:  View [dbo].[T7DQ48L]    Script Date: 01/04/2018 10:27:29 ******/
SET ANSI_NULLS ON
GO

SET QUOTED_IDENTIFIER ON
GO

create view [dbo].[T7DQ48L]
as
select t.fitemid id,19 DirID,convert(tinyint,'0') SecLevel,t4.FCreateUser Builder,--IMS系统数据创建人
case when t4.flastmoddate is null then t4.fcreatedate else t4.flastmoddate end LastModified,--IMS系统数据修改日期，调用物料主数据修改日期
case when t4.flastmoduser is null then t4.FCreateUser else t4.FLastModUser  end LastMender,--IMS系统数据更新人，调用物料主数据更新人
FNumber F1ZUO1H,--供应商编码
t.FName FSUEWNH,--供应商名称
FAddress FDSND8D,--地址
FContact FII59RM,--联系人
FMobilePhone FOSWR1Q,--移动电话
FPhone F0BKTFE,--电话
FFax F3P6HFN,--传真
FEmail FBKP3DS,--邮件地址
FValueAddRate FF2YU58,--增值税率
FBank F2SC7CI,--开户银行
FAccount FP35WIJ,--银行账号
case when FStatus=1072 then '使用' when FStatus=1073 then '未使用' when FStatus=1074 then '冻结' else null end FX7AY1K,--状态
FTaxNum F6EHFLB,--税务登记号
FLastTradeDate F3509VG,--最后交易日期
FlastTradeAmount F631V5Y,--最后交易金额
FLastReceiveDate FA8BCJP,--最后付款日期
FLastRPAmount FOAW5NN,--最后付款金额
t4.FCreateDate FFQB0S2,--创建日期
t4.FCreateUser F16D4U8,--创建人
t4.FLastModDate F0TE7XY,--修改日期
t4.FLastModUser F8DLL7T--修改人
from [AIS20130122132632].[dbo].t_supplier t
	INNER JOIN [AIS20130122132632].[dbo].t_BaseProperty t4 ON t4.FItemID = t.Fitemid and t4.ftypeid=3--调用创建人及更新日期
where t.fdeleted='0'
GO
