def loadDataSet(filename):
    """
    DataSet = []
    DataMat = []
    fileHandle = open(filename)    #    打开文件
    fileList = fileHandle.readlines()    #    将文件全部读入到fileList中
    for fileline in fileList:    #    逐行读取数据
        temp = fileline.strip('\n')    #    strip用于删除每一行尾端的换行符（'\n'）
        temp1 = temp.split('\t')    #    将每个数据根据'\t'分隔开
        DataSet .append(temp1)    #    添加到列表DataSet中    ********************（1）

    for line in DataSet :    #    遍历DataSet中每个列表
        for line_son in line:    #    遍历DataSet中每个列表中的数据
            temp2 = line_son.split(' ')    #    将每个列表中的数据利用' '进行分割
            for i in temp2:    #    遍历分割后的子列表
                if i != '':    #    如果不等于' '就添加到新列表Datamat中
                    DataMat .append(float(i))    #    将每个数据进行转换
    return(DataMat )    #    返回最终列表
