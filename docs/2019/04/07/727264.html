A=[1,1,1;
   1,3,9;
   1,7,49];
b=[6,5,2];
n=length(b);%方程个数n

% m=zeros(1,3);
%-------------消去-----------
for k=1:n-1
    if A(k,k)==0;
        error('Error');
    end
    
     for j=2:n
         if A(k,1)<A(j,1)
%                 tmp=A(1,1);A(1,1)=A(j,1);A(j,1)=tmp;
%                 tmp=A(1,2);A(1,2)=A(j,2);A(j,2)=tmp;
%                 tmp=A(1,3);A(1,3)=A(j,3);A(j,3)=tmp;
                tmp=A(k,:);A(k,:)=A(j,:);A(j,:)=tmp;
                tmp=b(1);b(1)=b(2);b(2)=tmp;
         end
     end
    
    for i=k+1:n
         mik=-A(i,k)/A(k,k);
         for j=k:n
              A(i,j)=A(i,j)+mik*A(k,j);
         end
%          A;
         b(i)=b(i)+mik*b(k);
    end
end
    
%-------------回代-----------
x(n)=b(n)/A(n,n);
for k=n-1:-1:1
    S=b(k);
    for j=k+1:n
        S=S-A(k,j)*x(j);
    end
    x(k)=S/A(k,k);
end
% x;         %程序的计算结果
z=linspace(0,16,500);
y=x(1)+x(2)*z+x(3)*power(z,2);
plot(z,y);

A=[1,1,1;
