        public static uint GetCRC32(byte[] bytes)
        {
            uint iCount = (uint)bytes.Length;
            uint crc = 0xFFFFFFFF;

            for (uint i = 0; i < iCount; i++)
            {                
                crc = (crc << 8) ^ crcTable[(crc >> 24) ^ bytes[i]];
            }            

            return crc;
        }
