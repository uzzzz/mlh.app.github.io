Description

小Y最近在一家金券交易所工作。该金券交易所只发行交易两种金券：A纪念券（以下简称A券）和 B纪念券（以下
简称B券）。每个持有金券的顾客都有一个自己的帐户。金券的数目可以是一个实数。每天随着市场的起伏波动，
两种金券都有自己当时的价值，即每一单位金券当天可以兑换的人民币数目。我们记录第 K 天中 A券 和 B券 的
价值分别为 AK 和 BK（元/单位金券）。为了方便顾客，金券交易所提供了一种非常方便的交易方式：比例交易法
。比例交易法分为两个方面：（a）卖出金券：顾客提供一个 [0,100] 内的实数 OP 作为卖出比例，其意义为：将
 OP% 的 A券和 OP% 的 B券 以当时的价值兑换为人民币；（b）买入金券：顾客支付 IP 元人民币，交易所将会兑
换给用户总价值为 IP 的金券，并且，满足提供给顾客的A券和B券的比例在第 K 天恰好为 RateK；例如，假定接

注意到，同一天内可以进行多次操作。小Y是一个很有经济头脑的员工，通过较长时间的运作和行情测算，他已经
知道了未来N天内的A券和B券的价值以及Rate。他还希望能够计算出来，如果开始时拥有S元钱，那么N天后最多能

Input

输入第一行两个正整数N、S，分别表示小Y能预知的天数以及初始时拥有的钱数。接下来N行，第K行三个实数AK、B

【提示】
1.输入文件可能很大，请采用快速的读入方式。
2.必然存在一种最优的买卖方案满足：
每次买进操作使用完所有的人民币；
每次卖出操作卖出所有的金券。

只有一个实数MaxProfit，表示第N天的操作结束时能够获得的最大的金钱数目。答案保留3位小数。

题解:

         显然每次肯定是全部买入或者全部卖出.

         设f[i]表示到第i天A卷的最多数量.

         写出式子来发现可以斜率优化.然后发现两个量都不单调.

         然后cdq分治或者平衡树维护凸包即可.

代码：

#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
#define N 100010
#define eps 1e-9
#define inf 1e20
using namespace std;
double f[N];
int n,st[N],top;
struct point{double x,y,a,b,k,r;int id;}p[N],q[N];
bool cmp(point a,point b){return a.k>b.k;}
double cal(int x,int y){
  if (!y) return -inf;
  if (fabs(p[x].x-p[y].x)<eps) return inf;
  else return (p[y].y-p[x].y)/(p[y].x-p[x].x);
}
void solve(int l,int r){
  if (l==r){
    f[l]=max(f[l],f[l-1]);
    p[l].y=f[l]/(p[l].a*p[l].r+p[l].b);
    p[l].x=p[l].y*p[l].r;
    return;
  } 
  int mid=(l+r)>>1,l1=l,l2=mid+1;
  for (int i=l;i<=r;i++)
    if (p[i].id<=mid) q[l1++]=p[i];
    else q[l2++]=p[i];
  for (int i=l;i<=r;i++) p[i]=q[i];
  solve(l,mid);top=0;
  for (int i=l;i<=mid;i++){
    while (top>1&&cal(st[top-1],st[top])<cal(st[top-1],i)+eps) top--;
    st[++top]=i;
  }
  st[++top]=0;int t=1;
  for (int i=mid+1;i<=r;i++){
    while (t<top&&cal(st[t],st[t+1])+eps>p[i].k) t++;
    f[p[i].id]=max(f[p[i].id],p[i].a*p[st[t]].x+p[i].b*p[st[t]].y);
  }
  solve(mid+1,r);
  l1=l;l2=mid+1;
  for (int i=l;i<=r;i++)
    if (((p[l1].x<p[l2].x||(fabs(p[l1].x-p[l2].x))<eps&&p[l1].y<p[l2].y)||l2>r)&&l1<=mid) q[i]=p[l1++];
    else q[i]=p[l2++];
  for (int i=l;i<=r;i++) p[i]=q[i]; 
}
int main(){
  scanf("%d%lf",&n,&f[0]);
  for (int i=1;i<=n;i++){
    scanf("%lf%lf%lf",&p[i].a,&p[i].b,&p[i].r);
    p[i].k=-p[i].a/p[i].b;p[i].id=i;
  } 
  sort(p+1,p+n+1,cmp);
  solve(1,n);
  printf("%.3lf\n",f[n]);
}
