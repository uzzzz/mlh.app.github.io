三. -race 检查是否有race condition 

/*
race例子

go run -race trace.go

Author: xcl
Date: 2015-11-22
*/

package main

import (
	"time"
)

var (
	g int
)

func main() {

	go func() {
		g++
	}()

	go func() {
		g = g + 2
	}()

	time.Sleep(1 * time.Second)

}

/*
